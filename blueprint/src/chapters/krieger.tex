\chapter{Generators: Shannon--McMillan--Breiman and Krieger's theorem}
\label{chap:krieger}

The Kolmogorov--Sinai entropy of the classical-entropy development measures the average
information produced by a transformation per unit time. This chapter documents the
\emph{generator} side of that theory (the 29 modules of \texttt{ErgodicTheory/Krieger/}): the
pointwise \emph{Shannon--McMillan--Breiman theorem} (entropy equipartition), the
\emph{Rokhlin--Kakutani tower lemma}, the symbolic coding stack built on them (name counting and
sentinel prefix codes), and the two generator headliners --- the countable finite-entropy
two-sided generator of Rokhlin, Keane--Serafin and Downarowicz, and \emph{Krieger's finite
generator theorem}: an ergodic aperiodic automorphism of entropy below \(\log k\) admits a
two-sided generating partition with at most \(k\) cells.

Throughout, \((\alpha,\mu)\) is a probability space, \(T\) a measure-preserving transformation,
and \(P\) a finite measurable partition indexed by a finite type \(\iota\) (a
\texttt{MeasurePartition}: measurable cells, pairwise a.e.-disjoint, covering \(\alpha\)). For the
coding and generator layers, \(e\) is a measure-preserving \emph{automorphism} (a measurable
equivalence \(\alpha \simeq \alpha\)) of a standard Borel probability space and iterates range
over \(\Z\). We write
\(\bigvee_{k=0}^{n-1} T^{-k}P\) for the \(n\)-fold iterated join, \(H(\cdot)\) for Shannon entropy,
and \(h(P,T)\) for the Kolmogorov--Sinai entropy of the pair. Every result below is formalized
sorry-free; the two headline generator theorems are stated relative to explicitly disclosed
hypothesis bundles (their honest formalization boundary, described in place).

\section{The information function and its chain rules}

\begin{definition}[Information function of the iterated join]\label{infoFun}
  \lean{ErgodicTheory.Krieger.infoFun}\leanok
  Every point \(x\) has an \(n\)-step \emph{itinerary}
  \lean{ErgodicTheory.Krieger.itinerary} \(f : \Fin n \to \iota\), recording for each
  \(k < n\) which cell of \(P\) contains \(T^k x\); among the admissible codes the
  \emph{least} one in a fixed enumeration is chosen, which makes the selector genuinely
  measurable (the cells may overlap on a null set, so a bare choice function would not be).
  The \emph{atom of \(x\)} is the cell \(\bigcap_{k<n} T^{-k}(P_{f(k)})\) of the iterated join
  containing \(x\), and the \emph{information function} is
  \[
    i_n(x) \;=\; -\log \mu\bigl(\mathrm{atom}_n(x)\bigr),
  \]
  the surprise of learning the \(n\)-name of \(x\). It is nonnegative and measurable
  (\lean{ErgodicTheory.Krieger.measurable_infoFun}): in its simple-function form it is a finite
  sum of indicators of the itinerary fibers, each the difference of a join cell and the finitely
  many strictly smaller cells.
\end{definition}

\begin{theorem}[The information function integrates to the join entropy]\label{integral_infoFun_eq}
  \lean{ErgodicTheory.Krieger.integral_infoFun_eq}\leanok
  \uses{infoFun, ksEntropyPartition}
  For every \(n\),
  \[
    \int_\alpha i_n \,d\mu \;=\; H\Bigl(\bigvee_{k=0}^{n-1} T^{-k}P\Bigr),
  \]
  the Shannon entropy of the \(n\)-fold iterated join (the \(n\)-th term of the
  Kolmogorov--Sinai entropy sequence).
\end{theorem}
\begin{proof}\leanok
  \uses{infoFun}
  Write \(i_n\) as the finite sum of indicators of the itinerary fibers weighted by
  \(-\log\mu(\text{cell})\) and integrate termwise. Each fiber differs from its join cell only
  inside the union of the \emph{other} cells --- a null set by pairwise a.e.-disjointness --- so
  fiber and cell have the same measure, and each summand becomes
  \(\mu(\text{cell})\cdot(-\log\mu(\text{cell})) = \operatorname{negMulLog}\mu(\text{cell})\),
  which summed over the cells is the join entropy.
\end{proof}

\begin{theorem}[Telescoped Breiman chain rule, entropy level]\label{ksEntropySeq_eq_sum_condEntropy}
  \lean{ErgodicTheory.Krieger.ksEntropySeq_eq_sum_condEntropy}\leanok
  \uses{ksEntropyPartition, condEntropyPartition}
  On a standard Borel space, the \(n\)-step join entropy telescopes into conditional entropies:
  \[
    H\Bigl(\bigvee_{k=0}^{n-1} T^{-k}P\Bigr)
      \;=\; \sum_{k=0}^{n-1}
        H\Bigl(P \,\Big|\, \sigma\bigl(T^{-1}\textstyle\bigvee_{j=0}^{k-1} T^{-j}P\bigr)\Bigr),
  \]
  where the \(k\)-th conditioning \(\sigma\)-algebra is generated by the cells of the
  \(T\)-pullback of the \(k\)-step join.
\end{theorem}
\begin{proof}\leanok
  \uses{ksEntropyPartition, condEntropyPartition}
  Induction on \(n\) from the one-step chain rule: reindexing along
  \(\Fin n \times \iota \simeq \Fin(n{+}1)\) (peeling the first symbol) identifies the
  \((n{+}1)\)-fold join with the join \(B \vee P\), \(B := T^{-1}(\bigvee_0^{n-1}T^{-k}P)\); the
  absolute chain rule \(H(B \vee P) = H(B) + H(P \mid B)\) and \(T\)-invariance of join entropy
  give \(H(\bigvee_0^{n}) = H(\bigvee_0^{n-1}) + H(P \mid \sigma(B))\), the conditional entropy
  bridged from its additive-over-cells form to the \(\sigma\)-algebra form.
\end{proof}

\begin{theorem}[The sharp KS rate as a conditional entropy]\label{ksEntropyPartition_eq_condEntropy_iSup}
  \lean{ErgodicTheory.Krieger.ksEntropyPartition_eq_condEntropy_iSup}\leanok
  \uses{ksEntropySeq_eq_sum_condEntropy, ksEntropyPartition, condEntropyPartition}
  On a standard Borel space (with \(\iota\) nonempty), the Kolmogorov--Sinai entropy of the pair
  equals the conditional entropy of \(P\) given the \(\sigma\)-algebra of the strict future:
  \[
    h(P,T) \;=\; H\Bigl(P \,\Big|\,
      \bigvee_{k}\sigma\bigl(T^{-1}\textstyle\bigvee_{j=0}^{k-1} T^{-j}P\bigr)\Bigr)
      \;=\; H\Bigl(P \,\Big|\, \sigma\Bigl(\bigvee_{j\ge 1} T^{-j}P\Bigr)\Bigr).
  \]
  This is the integral-level statement of the sharp SMB rate: the precise constant the pointwise
  theorem converges to. No ergodicity is needed.
\end{theorem}
\begin{proof}\leanok
  \uses{ksEntropySeq_eq_sum_condEntropy, condEntropyPartition, ksEntropyPartition}
  The conditioning \(\sigma\)-algebras increase in \(k\), so the fixed-partition L\'evy theorem
  (martingale convergence of conditional entropies) sends
  \(H(P \mid \sigma_k) \to H(P \mid \bigvee_k \sigma_k)\). By
  \Cref{ksEntropySeq_eq_sum_condEntropy}, \(\tfrac1n H(\bigvee_0^{n-1}T^{-k}P)\) is exactly the
  Ces\`aro average of that convergent sequence, hence tends to the same limit; matching it against
  the Fekete limit defining \(h(P,T)\) gives the identity.
\end{proof}

\section{The pointwise Shannon--McMillan--Breiman theorem}

\begin{theorem}[Crude name-count bound, Birkhoff-free]\label{ae_limsup_div_infoFun_le_log_card}
  \lean{ErgodicTheory.Krieger.ae_limsup_div_infoFun_le_log_card}\leanok
  \uses{infoFun}
  For a measure-preserving \(T\) and a finite partition indexed by a nonempty \(\iota\),
  \[
    \limsup_{n\to\infty} \frac{i_n(x)}{n} \;\le\; \log \#\iota
    \qquad\text{for \(\mu\)-a.e.\ } x.
  \]
  Neither ergodicity nor any ergodic theorem enters: this is the Algoet--Cover engine fed the
  uniform competing measure, whose partition-function bound
  \(\int \exp\bigl(i_n - n\log\#\iota\bigr)\,d\mu \le 1\) is the lemma
  \lean{ErgodicTheory.Krieger.lintegral_exp_infoFun_sub_log_card_le_one}.
\end{theorem}
\begin{proof}\leanok
  \uses{infoFun}
  On each itinerary fiber the integrand \(\exp(i_n - n\log\#\iota)\) is the constant
  \(\mu(\text{cell})^{-1}\cdot(\#\iota)^{-n}\) (or vanishes with the cell), and the join has at
  most \((\#\iota)^n\) non-null cells, giving the partition-function bound. Markov's inequality
  plus Borel--Cantelli then make \(\tfrac1n i_n \le \log\#\iota + \varepsilon\) eventually a.e.\
  for every \(\varepsilon > 0\).
\end{proof}

\begin{definition}[Conditional information function]\label{condInfoFun}
  \lean{ErgodicTheory.Krieger.condInfoFun}\leanok
  \uses{condEntropyPartition}
  For a sub-\(\sigma\)-algebra \(\mathcal A\) of a standard Borel space, the \emph{conditional
  information function} of \(P\) given \(\mathcal A\) is
  \[
    g_{\mathcal A}(x) \;=\; \sum_i \mathbf 1_{P_i}(x)\cdot
      \bigl(-\log \mu[P_i \mid \mathcal A](x)\bigr),
  \]
  the conditional probability realized by the regular conditional probability kernel; exactly one
  indicator survives at each point. Its integral is the conditional entropy: for
  \(\mathcal A \le m_\alpha\),
  \(\int g_{\mathcal A}\,d\mu = H(P \mid \mathcal A)\)
  (\lean{ErgodicTheory.Krieger.integral_condInfoFun_eq_condEntropy}); in particular
  \(g_{\mathcal A}\) is integrable.
\end{definition}

\begin{theorem}[Chung's maximal inequality]\label{chungTail}
  \lean{ErgodicTheory.Krieger.chungTail}\leanok
  \uses{condInfoFun}
  Let \(g_k\) be the conditional information function of \(P\) given the \(k\)-th Breiman
  conditioning \(\sigma\)-algebra
  \(\mathcal C_k = \sigma(T^{-1}\bigvee_{j=0}^{k-1}T^{-j}P)\), and let
  \(g^{*} = \sup_k g_k\) be the Chung maximal information function
  (\lean{ErgodicTheory.Krieger.condInfoMaxFun}, valued in \([0,\infty]\)). Then for every cell
  \(P_{i_0}\) and every \(\lambda > 0\),
  \[
    \mu\bigl(\{x \in P_{i_0} \mid g^{*}(x) > \lambda\}\bigr) \;\le\; e^{-\lambda}.
  \]
  Consequently \(g^{*} \in L^1(\mu)\), with
  \(\int g^{*}\,d\mu \le H(P) + 1\)
  (\lean{ErgodicTheory.Krieger.lintegral_condInfoMaxFun_lt_top}) --- Chung's \(L^1\) domination,
  the one genuinely analytic leaf of the SMB proof.
\end{theorem}
\begin{proof}\leanok
  \uses{condInfoFun}
  A stopping-time (first-passage) argument: stratify \(\{g^{*} > \lambda\}\cap P_{i_0}\) by the
  first level \(k\) at which the conditional probability of \(P_{i_0}\) drops below
  \(e^{-\lambda}\). Each stratum is \(\mathcal C_k\)-measurable, so its intersection with
  \(P_{i_0}\) has measure at most \(e^{-\lambda}\) times the stratum's measure by the defining
  property of conditional probability; summing the disjoint strata gives the tail bound.
  Integrability follows by the layer-cake formula: summing the per-cell tails yields
  \(\int g^{*} \le H(P) + 1\).
\end{proof}

\begin{theorem}[Pointwise SMB, from the Breiman telescoping]\label{ae_tendsto_div_infoFun}
  \lean{ErgodicTheory.Krieger.ae_tendsto_div_infoFun}\leanok
  \uses{condInfoFun, chungTail, ksEntropyPartition_eq_condEntropy_iSup}
  Let \(T\) be ergodic and let \((i_n)\) be any sequence of functions satisfying, for
  \(\mu\)-a.e.\ \(x\) and all \(n\), the Breiman telescoping identity
  \[
    i_n(x) \;=\; \sum_{j=0}^{n-1} g_{n-j}(T^{j}x),
  \]
  where \(g_k\) is the conditional information function of \(P\) given \(\mathcal C_k\). Then
  \[
    \frac{i_n(x)}{n} \;\longrightarrow\; h(P,T)
    \qquad\text{for \(\mu\)-a.e.\ } x.
  \]
\end{theorem}
\begin{proof}\leanok
  \uses{condInfoFun, chungTail, ksEntropyPartition_eq_condEntropy_iSup}
  Split \(\tfrac1n\sum_{j<n} g_{n-j}(T^jx)\) into the Birkhoff main term
  \(\tfrac1n\sum_{j<n} g_\infty(T^jx)\) and the Ces\`aro tail
  \(\tfrac1n\sum_{j<n}(g_{n-j}-g_\infty)(T^jx)\), where \(g_\infty\) is the conditional
  information function given the strict-future \(\sigma\)-algebra \(\bigvee_k \mathcal C_k\). The
  main term converges a.e.\ to \(\int g_\infty\,d\mu = h(P,T)\) by the pointwise ergodic theorem
  and the sharp rate identity (\Cref{ksEntropyPartition_eq_condEntropy_iSup}). The tail vanishes
  a.e.\ by the Maker/Breiman dominated-Ces\`aro argument
  (\lean{ErgodicTheory.Krieger.makerTail}): \(|g_{n-j}-g_\infty|\) is dominated by the integrable
  maximal function \(g^{*} + g_\infty\) of \Cref{chungTail}, and L\'evy downward convergence
  \(g_k \to g_\infty\) a.e.\ makes the dominated Birkhoff averages of the sup-tails vanish.
\end{proof}

\begin{theorem}[Pointwise Shannon--McMillan--Breiman theorem]\label{ae_tendsto_div_infoFun_self}
  \lean{ErgodicTheory.Krieger.ae_tendsto_div_infoFun_self}\leanok
  \uses{infoFun, ae_tendsto_div_infoFun, ksEntropyPartition}
  Let \(T\) be an ergodic measure-preserving transformation of a standard Borel probability space
  and \(P\) a finite measurable partition (nonempty index). Then the concrete information
  functions of \Cref{infoFun} satisfy
  \[
    \frac{i_n(x)}{n} \;\longrightarrow\; h(P,T)
    \qquad\text{for \(\mu\)-a.e.\ } x :
  \]
  the measure of the atom of \(x\) in the \(n\)-fold join decays like \(e^{-n\,h(P,T)}\)
  (entropy equipartition, the AEP of ergodic theory).
\end{theorem}
\begin{proof}\leanok
  \uses{infoFun, ae_tendsto_div_infoFun}
  The one-step factorization of the information weight (peeling the first symbol of the itinerary)
  telescopes, a.e., the concrete \(i_{n+1}(x)\) into the Breiman partial sum
  \(\sum_{j<n} g_{n-j}(T^jx)\) plus a single edge term \(g_0(T^n x)\). The partial sum converges
  after division by \(n\) via \Cref{ae_tendsto_div_infoFun}; the edge term dies since
  \(\tfrac1n g_0(T^n x) \to 0\) a.e.\ for the integrable \(g_0\); an index shift finishes.
\end{proof}

\begin{theorem}[In-measure upper equipartition]\label{upperSMBInMeasure_of_ergodic}
  \lean{ErgodicTheory.Krieger.upperSMBInMeasure_of_ergodic}\leanok
  \uses{ae_tendsto_div_infoFun_self}
  For ergodic \(T\), the in-measure SMB upper bound
  \lean{ErgodicTheory.Krieger.UpperSMBInMeasure} holds: for every \(\varepsilon > 0\),
  \[
    \mu\Bigl(\Bigl\{x \;\Big|\; h(P,T) + \varepsilon < \tfrac1N\, i_N(x)\Bigr\}\Bigr)
      \;\longrightarrow\; 0 \qquad (N \to \infty).
  \]
  This McMillan-type form is exactly what the name-count covering bound below consumes.
\end{theorem}
\begin{proof}\leanok
  \uses{ae_tendsto_div_infoFun_self}
  Almost-everywhere convergence (\Cref{ae_tendsto_div_infoFun_self}) makes the deviation sets
  eventually empty along a.e.\ orbit; convergence in measure of their indicators to the empty set
  follows on the finite measure space.
\end{proof}

\section{The Rokhlin tower lemma}

\begin{theorem}[Rokhlin--Kakutani tower lemma]\label{rokhlin_tower}
  \lean{ErgodicTheory.Krieger.rokhlin_tower}\leanok
  Let \(e\) be an ergodic measure-preserving automorphism of a standard Borel probability space
  with non-atomic \(\mu\). For every height \(N \ge 1\) and every \(\varepsilon > 0\) there is a
  measurable base \(B\) whose first \(N\) iterates \(B, eB, \dots, e^{N-1}B\) are pairwise
  disjoint and whose union covers all but \(\varepsilon\) of the space:
  \[
    \mu\Bigl(\bigcup_{i=0}^{N-1} e^{i}B\Bigr) \;>\; 1 - \varepsilon.
  \]
  (Ergodicity plus non-atomicity supplies the aperiodicity the classical statement assumes.)
\end{theorem}
\begin{proof}\leanok
  The Kakutani skyscraper construction. Pick a positive measurable set \(A\) with
  \(\mu(A) < \varepsilon/N\); by ergodicity a.e.\ point returns to \(A\), so the return-time
  levels of \(A\) tile the space mod \(0\). Grouping each return column into complete blocks of
  height \(N\) yields the base \(B\); the levels are disjoint by construction, and the uncovered
  part lies in the top incomplete floors of the columns, of total measure at most
  \(N\mu(A) < \varepsilon\).
\end{proof}

\section{The coding stack}

The passage from equipartition to a finite generator is combinatorial: count the names that carry
most of the mass, encode them into a slightly larger alphabet by a prefix-free (sentinel) code
painted along Rokhlin towers, and phrase ``generator'' as a mod-\(0\) statement about the two-sided
itinerary \(\sigma\)-algebra. We record one representative node for each layer.

\begin{theorem}[AEP covering bound: few names carry the mass]\label{exists_cover_names_card_le}
  \lean{ErgodicTheory.Krieger.exists_cover_names_card_le}\leanok
  \uses{infoFun, upperSMBInMeasure_of_ergodic, ksEntropyPartition}
  Let \(T\) be measure preserving and assume the in-measure SMB upper bound of
  \Cref{upperSMBInMeasure_of_ergodic}. Write \(h = h(P,T)\). For every \(\varepsilon > 0\) and all
  sufficiently large \(N\), there is a finite set \(S\) of rank-\(N\) names
  \(g : \Fin N \to \iota\) with
  \[
    \mu\Bigl(\bigcup_{g\in S}\;\bigcap_{k<N} T^{-k}P_{g(k)}\Bigr) \;\ge\; 1 - \varepsilon
    \qquad\text{and}\qquad
    \#S \;\le\; \bigl\lfloor e^{N(h+\varepsilon)} \bigr\rfloor.
  \]
  The set \(S\) is the set of \emph{good names} (\lean{ErgodicTheory.Krieger.goodNames}) --- names
  whose join cell has measure at least \(e^{-N(h+\varepsilon)}\); the cardinality bound is the
  unconditional pigeonhole \lean{ErgodicTheory.Krieger.card_goodNames_le_exp}.
\end{theorem}
\begin{proof}\leanok
  \uses{infoFun, upperSMBInMeasure_of_ergodic}
  Pigeonhole: cells of measure \(\ge e^{-N(h+\varepsilon)}\) number at most
  \(e^{N(h+\varepsilon)}\), since they are essentially disjoint in a probability space. Coverage:
  the good cells cover the set \(\{\tfrac1N i_N \le h+\varepsilon\}\), whose complement has
  measure \(< \varepsilon\) eventually by the in-measure SMB upper bound.
\end{proof}

\begin{theorem}[Sentinel prefix code]\label{exists_sentinelEncoding}
  \lean{ErgodicTheory.Krieger.exists_sentinelEncoding}\leanok
  Over an alphabet \(\Fin l\) with a reserved sentinel letter \(s\), any finite name set with
  \(\#\mathrm{Name} \le (l-1)^m\) admits an injection
  \(\mathrm{enc} : \mathrm{Name} \hookrightarrow \mathrm{List}(\Fin l)\) into blocks of length
  \(m+1\), each containing the sentinel exactly once (at its end). The blocks form a prefix-free
  family, so arbitrary concatenations are uniquely decodable
  (\lean{ErgodicTheory.Krieger.sentinelEncodeList_injective}); when
  \(e^{N(h+\varepsilon)} < k^{N}\), the good names of \Cref{exists_cover_names_card_le} embed into
  \(\Fin k\)-blocks of length \(O(N)\) --- the symbolic code from which the coding partition is
  read off along a Rokhlin tower.
\end{theorem}
\begin{proof}\leanok
  The name set embeds into the fixed-length data words \(\Fin m \to \Fin(l-1)\) over the
  non-sentinel letters by the cardinality bound; appending the sentinel produces blocks in which
  the sentinel marks exactly the block boundaries, so a straightforward take/drop induction
  recovers the block decomposition of any concatenation, giving injectivity.
\end{proof}

\begin{definition}[Two-sided generation mod 0]\label{IsGeneratingTwoSidedMod0}
  \lean{ErgodicTheory.Krieger.IsGeneratingTwoSidedMod0}\leanok
  The \emph{two-sided saturation} \lean{ErgodicTheory.Krieger.twoSidedSat} of a finite partition
  \(P\) under an automorphism \(e\) is the \(\sigma\)-algebra
  \(\bigvee_{n\in\Z} e^{-n}\sigma(P)\) generated by the two-sided \(P\)-itinerary. The partition
  \emph{two-sidedly generates \((\alpha, e, \mu)\) mod \(0\)} when the ambient \(\sigma\)-algebra
  is contained in the \(\mu\)-completion of the saturation --- every measurable set is recovered
  from the itinerary up to a null set. This mod-\(0\) form (not the literal equality of
  \(\sigma\)-algebras) is the faithful conclusion of Krieger's theorem. The
  \(\mathrm{Countable}\)-indexed analogue for a family of cells \(Q : \kappa \to
  \mathrm{Set}\,\alpha\) is \lean{ErgodicTheory.Krieger.IsGeneratingTwoSidedMod0c}.
\end{definition}

\begin{definition}[Countable Shannon entropy]\label{cHmu}
  \lean{ErgodicTheory.Krieger.cHμ}\leanok
  For a countable family of cells \(s : \iota \to \mathrm{Set}\,\alpha\), the \emph{countable
  Shannon entropy} is
  \(\;c H_\mu(s) = \sum_i \operatorname{negMulLog}\,\mu(s_i)
    = -\sum_i \mu(s_i)\log\mu(s_i)\),
  an unconditionally convergent \texttt{tsum}. Downarowicz's Fact 1.1.4 supplies the finiteness
  criterion (\lean{ErgodicTheory.Krieger.cHμ_summable_of_summable_index_mul}): if the
  index-weighted masses satisfy \(\sum_i i\cdot\mu(s_i) < \infty\), the entropy terms are
  summable, so \(cH_\mu(s)\) is a genuine finite sum. This is the form in which the
  Keane--Serafin construction certifies finite entropy of its limit partition.
\end{definition}

\section{The generator theorems}

\begin{theorem}[Countable finite-entropy two-sided generator (Rokhlin; Keane--Serafin)]\label{exists_countable_twoSided_generator}
  \lean{ErgodicTheory.Krieger.exists_countable_twoSided_generator}\leanok
  \uses{IsGeneratingTwoSidedMod0, cHmu}
  Let \((\alpha,\mu)\) be standard Borel with automorphism \(e\), and let
  \(Q : \N \to \mathrm{Set}\,\alpha\) be a countable family of measurable cells such that
  (i) the two-sided \(Q\)-itinerary recovers each set of a fixed standard-Borel generating
  sequence up to a \(\mu\)-null set, and (ii) the entropy terms
  \(i \mapsto \operatorname{negMulLog}\,\mu(Q_i)\) are summable. Then there is a countable family
  that two-sidedly generates \((\alpha,e,\mu)\) mod \(0\) and has finite static Shannon entropy
  --- sub-problem A of Krieger's theorem (Downarowicz, Thm.~4.2.3, first half). The structural
  reduction from the inductive Keane--Serafin data --- an increasing sequence of finite partitions
  with per-level recovery and a summable geometric mass envelope --- is
  \lean{ErgodicTheory.Krieger.exists_countable_twoSided_generator_of_levels}; the construction of
  that data from the raw dynamical hypotheses (the SMB-driven Rokhlin-tower marker painting of
  Keane--Serafin \S2) is the disclosed residual, isolated as an explicit hypothesis bundle rather
  than formalized.
\end{theorem}
\begin{proof}\leanok
  \uses{IsGeneratingTwoSidedMod0, cHmu}
  Recovering every set of a generating sequence mod \(0\) pushes the ambient \(\sigma\)-algebra
  into the \(\mu\)-completion of the two-sided saturation of \(Q\) (monotone-class/completion
  bookkeeping on the countable saturation), which is mod-\(0\) two-sided generation; the entropy
  summability is carried verbatim. In the levels form, the per-level families are enumerated
  through the pairing bijection \(\N \simeq \N\times\N\), the per-level recovery lifts to the
  union family, and the geometric envelope certifies summability via
  \Cref{cHmu}.
\end{proof}

\begin{definition}[Krieger coding data]\label{KriegerCodingData}
  \lean{ErgodicTheory.Krieger.KriegerCodingData}\leanok
  \uses{IsGeneratingTwoSidedMod0}
  For an automorphism \(e\) and \(k \in \N\), a \emph{Krieger coding datum} bundles: the
  measure-preservation of \(e\); a countable index type \(\kappa\) with measurable cells
  \(Q : \kappa \to \mathrm{Set}\,\alpha\) two-sidedly generating mod \(0\); and a
  \(\Fin k\)-valued partition \(P\) that \emph{codes} \(Q\) two-sidedly mod \(0\)
  (\lean{ErgodicTheory.Krieger.CodesTwoSidedMod0c}): the two-sided \(P\)-itinerary recovers each
  \(Q\)-cell up to a \(\mu\)-null set. This is exactly the object the Krieger construction
  produces when \(\log k\) exceeds the entropy: \(Q\) from
  \Cref{exists_countable_twoSided_generator}, and \(P\) from the column coding of the
  \(\le k^{N}\) good names (\Cref{exists_cover_names_card_le} and
  \Cref{exists_sentinelEncoding}) along refining Rokhlin towers (\Cref{rokhlin_tower}).
\end{definition}

\begin{theorem}[Recovery assembly of Krieger's theorem]\label{krieger_finite_generator_of_coding}
  \lean{ErgodicTheory.Krieger.krieger_finite_generator_of_coding}\leanok
  \uses{KriegerCodingData, IsGeneratingTwoSidedMod0}
  Given a Krieger coding datum \(D\) for \((e, \mu, k)\), there exists a partition
  \(P\) into at most \(k\) cells that two-sidedly generates \((\alpha, e, \mu)\) mod \(0\). No
  entropy, ergodicity or aperiodicity hypotheses enter: those are consumed upstream, in producing
  \(D\). This is the pure recovery content of Krieger's theorem.
\end{theorem}
\begin{proof}\leanok
  \uses{KriegerCodingData, IsGeneratingTwoSidedMod0}

\begin{theorem}[Krieger's finite generator theorem]\label{krieger_finite_generator}
  \lean{ErgodicTheory.Krieger.krieger_finite_generator}\leanok
  \uses{krieger_finite_generator_of_coding, KriegerCodingData, ksEntropy}
  Let \(e\) be an ergodic, aperiodic (\lean{ErgodicTheory.Krieger.Aperiodic}: every set of
  \(n\)-periodic points, \(n \ne 0\), is \(\mu\)-null), measure-preserving automorphism of a
  probability space, and let \(k \in \N\) satisfy
  \[
    h(e) \;<\; \log k,
  \]
  where \(h(e)\) is the Kolmogorov--Sinai entropy of the system. If the Krieger coding
  construction supplies a coding datum (\Cref{KriegerCodingData}), then \(e\) admits a
  \emph{finite two-sided generator of size \(\le k\), mod \(0\)}: a partition
  \(P\) indexed by \(\Fin k\) with \(\mathrm{IsGeneratingTwoSidedMod0}\ e\ P\). The
  entropy threshold is pinned to the genuine \(h(e)\), so the hypothesis is a real entropy
  constraint. \emph{Interface disclosure:} the coding datum has no in-repo constructor --- the
  tower-painting combinatorics that would discharge it from the dynamical hypotheses alone are not
  formalized --- so this is the honest conditional form of Krieger's theorem, with the dynamical
  hypotheses retained as the faithful classical interface that the upstream layers
  (\Cref{rokhlin_tower}, \Cref{exists_cover_names_card_le}, \Cref{exists_sentinelEncoding} and
  \Cref{exists_countable_twoSided_generator}) are designed to feed.
\end{theorem}
\begin{proof}\leanok
  \uses{krieger_finite_generator_of_coding}
  Immediate from the recovery assembly \Cref{krieger_finite_generator_of_coding} applied to the
  supplied coding datum, which already carries the measure-preservation the cross-layer recovery
  needs.
\end{proof}
